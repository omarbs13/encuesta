CREATE TYPE TRepuesta AS TABLE
(	
	[cRespuesta] [nvarchar](300) NOT NULL,
	[dFecha] [date] NOT NULL,
	[fkSubPregunta] [int] NOT NULL,
	[cEmail] [varchar](max) NULL	
)



CREATE PROCEDURE dbo. usp_InsertResp
   @TVP TRepuesta READONLY
      AS
      SET NOCOUNT ON
      INSERT INTO Respuesta
         (
            cRespuesta
            , dFecha
            , fkSubPregunta
            , cEmail
         )
      SELECT *
      FROM @TVP;
